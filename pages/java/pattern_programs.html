2. Print a pyramid pattern of stars

            public class PyramidPattern {
                public static void printPyramid(int n) {
                    for (int i = 1; i <= n; i++) {
                        for (int j = 1; j <= n - i; j++) {
                            System.out.print(" "); // Print spaces for alignment
                        }
                        for (int k = 1; k <= 2 * i - 1; k++) {
                            System.out.print("*"); // Print stars to form pyramid
                        }

                    }
                }
            }

8. Print a Pascal's Triangle pattern 

            // Java program to print Pascal's Triangle
            public class PascalsTriangle {
                public static void printPascalsTriangle(int n) {
                    for (int i = 0; i < n; i++) {
                        int number = 1;
                        for (int j = 0; j < n - i; j++) {
                            System.out.print(" "); // Print leading spaces
                        }
                        for (int k = 0; k <= i; k++) {
                            System.out.print(number + " "); // Print the number in Pascal's Triangle
                            number = number * (i - k) / (k + 1); // Calculate next number
                        }
                        System.out.println(); // Move to the next line after each row
                    }
                }
            }

                        for (int j = 1; j <= n; j++) {

                            } else {
                                System.out.print(" ");
                            }
                        }

20. Print a square spiral pattern of numbers 

            // Java program to print a square spiral pattern of numbers
            public class SpiralPattern {
                public static void printSpiralPattern(int n) {
                    int[][] arr = new int[n][n];
                    int value = 1;
                    int top = 0, bottom = n - 1, left = 0, right = n - 1;
                    
                    while (value <= n * n) {
                        for (int i = left; i <= right; i++) arr[top][i] = value++;
                        top++;
                        for (int i = top; i <= bottom; i++) arr[i][right] = value++;
                        right--;
                        for (int i = right; i >= left; i--) arr[bottom][i] = value++;
                        bottom--;
                        for (int i = bottom; i >= top; i--) arr[i][left] = value++;
                        left++;
                    }
                    // Print the spiral pattern
                    for (int[] row : arr) {
                        for (int num : row) {
                            System.out.print(num + "\t");
                        }
                        System.out.println();
                    }
                }
            }
