2. Check if a string is a palindrome 

            // Java program to check if a string is a palindrome
            public class PalindromeCheck {
                public static boolean isPalindrome(String str) {
                    int start = 0, end = str.length() - 1;
                    while (start < end) {
                        if (str.charAt(start) != str.charAt(end)) {
                            return false; // Not a palindrome if characters don't match
                        }
                        start++;
                        end--;
                    }
                    return true; // String is a palindrome if we reach here
                }
            }
3. Count vowels and consonants in a string 

            // Java program to count vowels and consonants in a string
            public class VowelConsonantCount {
                public static int[] countVowelsAndConsonants(String str) {
                    int vowels = 0, consonants = 0;
                    str = str.toLowerCase(); // Convert to lowercase for easy checking
                    for (char c : str.toCharArray()) {
                        if ("aeiou".indexOf(c) != -1) {
                            vowels++; // Increment vowels if character is a vowel
                        } else if (c >= 'a' && c <= 'z') {
                            consonants++; // Increment consonants for other letters
                        }
                    }
                    return new int[]{vowels, consonants}; // Return count of vowels and consonants
                }
            }

5. Check if two strings are anagrams 

            // Java program to check if two strings are anagrams
            import java.util.Arrays;
            public class AnagramCheck {
                public static boolean areAnagrams(String str1, String str2) {
                    if (str1.length() != str2.length()) return false; // Not anagrams if lengths differ
                    char[] arr1 = str1.toCharArray();
                    char[] arr2 = str2.toCharArray();
                    Arrays.sort(arr1);
                    Arrays.sort(arr2);
                    return Arrays.equals(arr1, arr2); // True if sorted arrays match
                }
            }
6. Find all permutations of a string 

            // Java program to find all permutations of a string
            public class StringPermutations {
                public static void permute(String str, String prefix) {
                    if (str.length() == 0) {
                        System.out.println(prefix); // Print permutation when no characters are left
                    } else {
                        for (int i = 0; i < str.length(); i++) {
                            String rem = str.substring(0, i) + str.substring(i + 1);
                            permute(rem, prefix + str.charAt(i)); // Recursive call for permutation
                        }
                    }
                }
            }
7. Remove duplicate characters from a string 

            // Java program to remove duplicate characters from a string
            import java.util.HashSet;
            public class RemoveDuplicates {
                public static String removeDuplicates(String str) {
                    HashSet seen = new HashSet<>();
                    StringBuilder result = new StringBuilder();
                    for (char c : str.toCharArray()) {
                        if (!seen.contains(c)) {
                            seen.add(c);
                            result.append(c); // Append character if not seen before
                        }
                    }
                    return result.toString(); // Return string without duplicates
                }
            }

11. Convert a string to title case 

            // Java program to convert a string to title case
            public class TitleCase {
                public static String toTitleCase(String str) {
                    String[] words = str.toLowerCase().split("\\s+");
                    StringBuilder titleCase = new StringBuilder();
                    for (String word : words) {
                        if (word.length() > 0) {
                            titleCase.append(Character.toUpperCase(word.charAt(0))) // Capitalize first letter
                                     .append(word.substring(1)) // Append remaining letters
                                     .append(" ");
                        }
                    }
                    return titleCase.toString().trim(); // Return title-cased string
                }
            }

17. Reverse each word in a sentence

                public static String reverseWords(String sentence) {
                    String[] words = sentence.split(" ");
                    StringBuilder reversedSentence = new StringBuilder();
                    for (String word : words) {

                }
            }

                    LinkedHashMap charCount = new LinkedHashMap<>();
                    for (char c : str.toCharArray()) {
                        charCount.put(c, charCount.getOrDefault(c, 0) + 1); // Count each character
                    }

                    }

                }
            }

21. Find the longest substring without repeating characters 

            // Java program to find the longest substring without repeating characters
            import java.util.HashSet;
            
            public class LongestUniqueSubstring {
                public static String longestUniqueSubstring(String str) {
                    String longestSubstring = "";
                    for (int i = 0; i < str.length(); i++) {
                        HashSet seenChars = new HashSet<>();
                        StringBuilder currentSubstring = new StringBuilder();
                        for (int j = i; j < str.length(); j++) {
                            char ch = str.charAt(j);
                            if (seenChars.contains(ch)) break; // Stop if character is repeated
                            seenChars.add(ch);
                            currentSubstring.append(ch);
                        }
                        if (currentSubstring.length() > longestSubstring.length()) {
                            longestSubstring = currentSubstring.toString(); // Update longest substring
                        }
                    }
                    return longestSubstring;
                }
            }

            public class MostFrequentCharacter {
                public static char mostFrequentCharacter(String str) {
                    HashMap charCount = new HashMap<>();
                    for (char c : str.toCharArray()) {
                        charCount.put(c, charCount.getOrDefault(c, 0) + 1); // Count each character
                    }
                    char mostFrequent = str.charAt(0);
                    int maxCount = 0;
                    for (char c : charCount.keySet()) {
                        if (charCount.get(c) > maxCount) {
                            maxCount = charCount.get(c);
                            mostFrequent = c; // Update most frequent character
                        }
                    }
                    return mostFrequent;
                }
            }

26. Find the length of the longest palindromic substring 

            // Java program to find the length of the longest palindromic substring
            public class LongestPalindromicSubstring {
                public static String longestPalindrome(String str) {
                    String longest = "";
                    for (int i = 0; i < str.length(); i++) {
                        String oddPalindrome = expandAroundCenter(str, i, i); // Odd length palindromes
                        String evenPalindrome = expandAroundCenter(str, i, i + 1); // Even length palindromes
                        if (oddPalindrome.length() > longest.length()) longest = oddPalindrome;
                        if (evenPalindrome.length() > longest.length()) longest = evenPalindrome;
                    }
                    return longest;
                }
            
                private static String expandAroundCenter(String str, int left, int right) {
                    while (left >= 0 && right < str.length() && str.charAt(left) == str.charAt(right)) {
                        left--;
                        right++;
                    }
                    return str.substring(left + 1, right);
                }
            }
