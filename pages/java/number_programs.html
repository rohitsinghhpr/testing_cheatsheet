1. Check if a number is prime 

        // Java program to check if a number is prime
        public class PrimeCheck {
            public static boolean isPrime(int num) {
                // Edge cases: numbers less than 2 are not prime
                if (num < 2) return false;
                for (int i = 2; i <= Math.sqrt(num); i++) {
                    // If num is divisible by any number other than 1 and itself, it's not prime
                    if (num % i == 0) return false;
                }
                return true; // Number is prime if we reach here
            }
        }
2. Find the factorial of a number 

        // Java program to find the factorial of a number
        public class Factorial {
            public static int factorial(int n) {
                int result = 1;
                for (int i = 1; i <= n; i++) {
                    result *= i; // Multiply result by i in each iteration
                }
                return result; // Return the computed factorial
            }
        }
3. Check if a number is an Armstrong number 

        // Java program to check if a number is an Armstrong number
        public class ArmstrongNumber {
            public static boolean isArmstrong(int num) {
                int original = num, sum = 0;
                int digits = String.valueOf(num).length(); // Calculate the number of digits
                while (num > 0) {
                    int digit = num % 10;
                    sum += Math.pow(digit, digits); // Add the power of the digit to the sum
                    num /= 10; // Remove the last digit
                }
                return sum == original; // True if sum equals original number
            }
        }

5. Reverse a number 

        // Java program to reverse a number
        public class ReverseNumber {
            public static int reverse(int num) {
                int reversed = 0;
                while (num != 0) {
                    int digit = num % 10; // Extract the last digit
                    reversed = reversed * 10 + digit; // Shift and add the digit
                    num /= 10; // Remove the last digit
                }
                return reversed; // Return the reversed number
            }
        }

        // Java program to check if a number is palindrome
        public class PalindromeCheck {
            public static boolean isPalindrome(int num) {
                int reversed = 0, original = num;
                while (num != 0) {
                    int digit = num % 10; // Extract the last digit
                    reversed = reversed * 10 + digit; // Construct the reversed number
                    num /= 10; // Remove the last digit
                }
                return original == reversed; // True if original equals reversed
            }
        }
7. Find the greatest common divisor (GCD) of two numbers 

        // Java program to find the GCD of two numbers
        public class GCD {
            public static int findGCD(int a, int b) {
                while (b != 0) {
                    int temp = b;
                    b = a % b; // Find remainder
                    a = temp; // Update a to be b
                }
                return a; // Return the GCD
            }
        }

            public static boolean isPerfect(int num) {
                int sum = 0;
                for (int i = 1; i <= num / 2; i++) {
                    if (num % i == 0) {

11. Find the LCM of two numbers 

            // Java program to find the LCM of two numbers
            public class LCM {
                public static int findLCM(int a, int b) {
                    return (a * b) / findGCD(a, b); // Use formula: LCM(a, b) = (a * b) / GCD(a, b)
                }
            
                private static int findGCD(int a, int b) {
                    while (b != 0) {
                        int temp = b;
                        b = a % b;
                        a = temp;
                    }
                    return a;
                }
            }
12. Check if a number is a strong number 

            // Java program to check if a number is a strong number
            // A strong number is a number where the sum of the factorial of each digit is equal to the number itself.
            public class StrongNumber {
                public static boolean isStrong(int num) {
                    int original = num, sum = 0;
                    while (num > 0) {
                        int digit = num % 10;
                        sum += factorial(digit); // Sum of factorial of each digit
                        num /= 10;
                    }
                    return sum == original; // True if sum equals original number
                }
            
                private static int factorial(int n) {
                    int fact = 1;
                    for (int i = 1; i <= n; i++) {
                        fact *= i;
                    }
                    return fact;
                }
            }

14. Sum of all prime numbers up to N 

            // Java program to calculate the sum of all prime numbers up to N
            public class SumOfPrimes {
                public static int sumOfPrimes(int n) {
                    int sum = 0;
                    for (int i = 2; i <= n; i++) {
                        if (isPrime(i)) sum += i; // Add prime numbers to sum
                    }
                    return sum;
                }
            
                private static boolean isPrime(int num) {
                    if (num < 2) return false;
                    for (int i = 2; i <= Math.sqrt(num); i++) {
                        if (num % i == 0) return false;
                    }
                    return true;
                }
            }

17. Check if a number is a palindrome using recursion 

            // Java program to check if a number is palindrome using recursion
            public class PalindromeRecursive {
                public static boolean isPalindrome(int num) {
                    return num == reverse(num, 0);
                }
            
                private static int reverse(int num, int rev) {
                    if (num == 0) return rev;
                    rev = (rev * 10) + (num % 10); // Construct reversed number
                    return reverse(num / 10, rev); // Recursive call with remaining digits
                }
            }

20. Find the largest digit in a number 

            // Java program to find the largest digit in a number
            public class LargestDigit {
                public static int largestDigit(int num) {
                    int maxDigit = 0;
                    while (num != 0) {
                        int digit = num % 10; // Get last digit
                        if (digit > maxDigit) maxDigit = digit; // Update max if found a larger digit
                        num /= 10; // Remove the last digit
                    }
                    return maxDigit; // Return the largest digit found
                }
            }
21. Check if a number is a Harshad number 

            // Java program to check if a number is a Harshad number
            // A Harshad number (or Niven number) is an integer that is divisible by the sum of its digits.
            public class HarshadNumber {
                public static boolean isHarshad(int num) {
                    int sum = 0, original = num;
                    while (num != 0) {
                        sum += num % 10; // Sum the digits
                        num /= 10; // Remove the last digit
                    }
                    return original % sum == 0; // True if original is divisible by sum of digits
                }
            }
22. Calculate the digital root of a number 

            // Java program to calculate the digital root of a number
            // The digital root of a number is the sum of its digits until a single-digit number is obtained.
            public class DigitalRoot {
                public static int digitalRoot(int num) {
                    while (num >= 10) {
                        int sum = 0;
                        while (num != 0) {
                            sum += num % 10; // Sum the digits
                            num /= 10; // Remove the last digit
                        }
                        num = sum; // Update num with the sum of digits
                    }
                    return num; // Return the single-digit result
                }
            }
