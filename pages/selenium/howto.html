Handling multiselect select dropdown -

Handling a multi-select dropdown in Selenium involves selecting multiple options from a dropdown menu that allows multiple selections. Here's an example of how you can handle a multi-select dropdown in Selenium using Java:

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.support.ui.Select;

public class MultiSelectDropdownExample {
public static void main(String[] args) {
WebDriver driver = new ChromeDriver();

// Open a webpage with a multi-select dropdown
driver.get("<https://www.example.com>");

// Find the multi-select dropdown element
WebElement multiSelectDropdown = driver.findElement(By.id("multi-select-dropdown"));

// Create a Select object for the multi-select dropdown
Select select = new Select(multiSelectDropdown);

// Check if the dropdown allows multiple selections
if (select.isMultiple()) {
// Select multiple options using selectByIndex, selectByValue, or selectByVisibleText methods
select.selectByIndex(0); // Index of the first option
select.selectByValue("value1"); // Value attribute of the option
select.selectByVisibleText("Option 3"); // Visible text of the option

// Deselect options if needed
select.deselectByIndex(1); // Index of the second option
select.deselectByValue("value2"); // Value attribute of the option
select.deselectByVisibleText("Option 4"); // Visible text of the option

// Get all selected options
System.out.println("Selected options:");
for (WebElement option : select.getAllSelectedOptions()) {
    System.out.println(option.getText());
}

// Get the first selected option
WebElement firstSelectedOption = select.getFirstSelectedOption();
System.out.println("First selected option: " + firstSelectedOption.getText());
} else {
System.out.println("The dropdown does not allow multiple selections.");
}

// Close the browser
driver.quit();
}
}

In this example:

  1. We open a webpage containing a multi-select dropdown.
  1. We find the multi-select dropdown element using its ID.
  1. We create a Select object for the multi-select dropdown element.
  1. We check if the dropdown allows multiple selections using isMultiple method.
  1. If the dropdown allows multiple selections, we select options using selectByIndex, selectByValue, or selectByVisibleText methods.
  1. We can also deselect options if needed using deselectByIndex, deselectByValue, or deselectByVisibleText methods.
  1. We retrieve selected options using getAllSelectedOptions and get the first selected option using getFirstSelectedOption.

Make sure to replace "<https://www.example.com>" with the URL of the webpage containing the multi-select dropdown. Adjust the locators (By.id, By.className, etc.) based on your element's attributes.

5. Checking for Broken Links

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

import java.io.IOException;
import java.net.HttpURLConnection;
import java.net.URL;
import java.util.List;

public class CheckBrokenLinksExample {
public static void main(String[] args) throws IOException {
System.setProperty("webdriver.chrome.driver", "path/to/chromedriver");

WebDriver driver = new ChromeDriver();
driver.get("<https://example.com>");

// Find all links on the page
List<WebElement> links = driver.findElements(By.tagName("a"));

for (WebElement link : links) {
String url = link.getAttribute("href");
if (url != null && !url.isEmpty()) {
    HttpURLConnection httpURLConnection = (HttpURLConnection) (new URL(url).openConnection());
    httpURLConnection.setRequestMethod("HEAD");
    httpURLConnection.connect();
    int responseCode = httpURLConnection.getResponseCode();
    if (responseCode >= 400) {
        System.out.println("Broken link: " + url + " - " + responseCode);
    } else {
        System.out.println("Valid link: " + url + " - " + responseCode);
    }
}
}

driver.quit();
}

}
